\usetikzlibrary{positioning}
\usetikzlibrary{shadows}

\tikzstyle{every entity} = [top color=white, bottom color=orange!80,
                            draw=yellow!50!black!100, drop shadow]

\centering
\scalebox{.87}{
\begin{tikzpicture}[node distance=1.5cm, every edge/.style={link}]

  %
  \node[entity]  (pro) {Proveedores}; % Talleres mecanicos - Services

% Contabilidad
  \node[entity]  (nts-cre) [right=1cm of pro] {Notas de Credito};
  \node[entity]  (fac)     [below=of pro] {Facturas};
  \node[entity]  (cli)     [right=of fac] {Clientes};
  \node[entity]  (rec)     [right=of cli] {Recibos};

  \node[entity]  (via)     [below=of cli]      {Viajes};
  \node[entity]  (via-des) [below=of via]      {Viajes - Destinos};
  \node[entity]  (des)     [below=of via-des]  {Destinos};

  % Transporte
  \node[entity]  (veh)     [below=of des]  {Vehiculos};
  \node[entity]  (tip-veh) [left=of veh]   {Tipos de Vehiculos};
  %
  \node[entity]  (cho-veh) [right=of veh]     {Choferes - Vehiculos};
  \node[entity]  (cho)     [right=2.2cm of des]     {Choferes};
  %
  \node[entity]  (car)     [left=of des]      {Cargas};
  \node[entity]  (tip-car) [left=of car]      {Tipos de Carga};


  % Relations
  \draw[link] (nts-cre) edge [<<->] (cli);
  \draw[link] (fac)     edge [<<->] (cli);
  \draw[link] (rec)     edge [<<->] (cli);
  \draw[link] (via)     edge [<<->] (cli);

  \draw[link] (via)     edge [<->>] (via-des);
  \draw[link] (via-des) edge [<<->] (des);

  \draw[link] (tip-car) edge [<->>] (car);
  \draw[link] (car)     edge [<<->] (des);
  \draw[link] (des)     edge [<<->] (cho);
  \draw[link] (des)     edge [<->]  (veh);

  \draw[link] (tip-veh) edge [<->>] (veh);
  \draw[link] (veh)     edge [<->>] (cho-veh);
  \draw[link] (cho-veh) edge [<<->] (cho);

  \end{tikzpicture}
}
